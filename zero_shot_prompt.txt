/**
 * Convert text to `snake_case`.
 *
 * The function handles both delimiter-based and camelCase inputs by:
 * - inserting underscores between lowercase-to-uppercase transitions
 * - converting spaces and hyphens to underscores
 * - lowercasing the final result
 *
 * This makes it suitable for values like `camelCase`, `Title Case`, and
 * `kebab-case`.
 *
 * @param {string} text Input text to normalize.
 * @returns {string} Snake_case representation of `text`.
 * @example
 * toSnakeCase('helloWorld'); // 'hello_world'
 * @example
 * toSnakeCase('User Profile-Name'); // 'user_profile_name'
 */
function toSnakeCase(text) {
    return text
        .replace(/([a-z])([A-Z])/g, '$1_$2')
        .replace(/[\s\-]+/g, '_')
        .toLowerCase();
}

module.exports = toSnakeCase;
